60 template <unsigned N>
61 FixedHash<N> jsToFixed(std::string const& _s) {
62  if (_s.substr(0, 2) == "0x")
63  // Hex
64  return FixedHash<N>(_s.substr(2 + std::max<unsigned>(N * 2, _s.size() - 2) - N * 2));
65  else if (_s.find_first_not_of("0123456789") == std::string::npos)
66  // Decimal
67  return (typename FixedHash<N>::Arith)(_s);
68  else
69  // Binary
70  return FixedHash<N>(); // FAIL
71 }
