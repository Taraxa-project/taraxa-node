59template <unsigned N>
60FixedHash<N> jsToFixed(std::string const& _s) {
61 if (_s.substr(0, 2) == "0x")
62 // Hex
63 return FixedHash<N>(_s.substr(2 + std::max<unsigned>(N * 2, _s.size() - 2) - N * 2));
64 else if (_s.find_first_not_of("0123456789") == std::string::npos)
65 // Decimal
66 return (typename FixedHash<N>::Arith)(_s);
67 else
68 // Binary
69 return FixedHash<N>(); // FAIL
70}

92inline uint64_t jsToInt(std::string const& _s) {
93 if (_s.substr(0, 2) == "0x") {
94 return std::stoull(_s, 0, 16); // Hex
95 } else if (_s.find_first_not_of("0123456789") == std::string::npos) {
96 return std::stoull(_s, 0, 10); // Decimal
97 }
98 // Still try to parse it as hex. Will throw the std::invalid_argument if it isn't possible
99 return std::stoull(_s, 0, 16);
100}
