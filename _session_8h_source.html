50 void send(std::string capability_name, unsigned packet_type, bytes payload, std::function<void()>&& on_done = {}) {
51 ba::post(m_socket->ref().get_executor(),
52 [=, this, _ = shared_from_this(), capability_name = std::move(capability_name),
53 payload = std::move(payload), on_done = std::move(on_done)]() mutable {
54 auto cap_itr = m_capabilities.find(capability_name);
55 assert(cap_itr != m_capabilities.end());
56 auto header = packet_type + cap_itr->second.offset;
57 assert(header <= std::numeric_limits<byte>::max());
58 bytes msg(1 + payload.size());
59 msg[0] = header;
60 memmove(msg.data() + 1, payload.data(), payload.size());
61 send_(std::move(msg), std::move(on_done));
62 });
63 }
