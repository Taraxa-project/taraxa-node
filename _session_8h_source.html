49 void send(std::string capability_name, unsigned packet_type, bytes payload, std::function<void()>&& on_done = {}) {
50 ba::post(m_socket->ref().get_executor(),
51 [=, this, _ = shared_from_this(), capability_name = std::move(capability_name),
52 payload = std::move(payload), on_done = std::move(on_done)]() mutable {
53 auto cap_itr = m_capabilities.find(capability_name);
54 assert(cap_itr != m_capabilities.end());
55 auto header = packet_type + cap_itr->second.offset;
56 assert(header <= std::numeric_limits<byte>::max());
57 bytes msg(1 + payload.size());
58 msg[0] = header;
59 memmove(msg.data() + 1, payload.data(), payload.size());
60 send_(std::move(msg), std::move(on_done));
61 });
62 }
