53  void send(std::string capability_name, unsigned packet_type, bytes payload, std::function<void()>&& on_done = {}) {
54  ba::post(m_socket->ref().get_executor(),
55  [=, this, _ = shared_from_this(), capability_name = std::move(capability_name),
56  payload = std::move(payload), on_done = std::move(on_done)]() mutable {
57  auto cap_itr = m_capabilities.find(capability_name);
58  assert(cap_itr != m_capabilities.end());
59  auto header = packet_type + cap_itr->second.offset;
60  assert(header <= std::numeric_limits<byte>::max());
61  bytes msg(1 + payload.size());
62  msg[0] = header;
63  memmove(msg.data() + 1, payload.data(), payload.size());
64  send_(std::move(msg), std::move(on_done));
65  });
66  }
