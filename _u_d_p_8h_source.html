165template <typename Handler, unsigned MaxDatagramSize>
166void UDPSocket<Handler, MaxDatagramSize>::connect() {
167 bool expect = false;
168 if (!m_started.compare_exchange_strong(expect, true)) return;
169
170 m_socket.open(bi::udp::v4());
171 try {
172 m_socket.bind(m_endpoint);
173 } catch (...) {
174 m_socket.bind(bi::udp::endpoint(bi::udp::v4(), m_endpoint.port()));
175 }
176
177 // clear write queue so reconnect doesn't send stale messages
178 Guard l(x_sendQ);
179 m_sendQ.clear();
180
181 m_closed = false;
182 doRead();
183}

185template <typename Handler, unsigned MaxDatagramSize>
186bool UDPSocket<Handler, MaxDatagramSize>::send(UDPDatagram const& _datagram) {
187 if (m_closed) return false;
188
189 Guard l(x_sendQ);
190 m_sendQ.push_back(_datagram);
191 if (m_sendQ.size() == 1) doWrite();
192
193 return true;
194}

196template <typename Handler, unsigned MaxDatagramSize>
197void UDPSocket<Handler, MaxDatagramSize>::doRead() {
198 if (m_closed) return;
199
200 auto self(UDPSocket<Handler, MaxDatagramSize>::shared_from_this());
201 m_socket.async_receive_from(
202 boost::asio::buffer(m_recvData), m_recvEndpoint,
203 boost::asio::bind_executor(strand_, [this, self](boost::system::error_code _ec, size_t _len) {
204 if (m_closed) return disconnectWithError(_ec);
205
206 if (_ec != boost::system::errc::success)
207 net_logger_->debug("Receiving UDP message failed. {} : {}", _ec.value(), _ec.message());
208
209 if (_len) m_host.onPacketReceived(this, m_recvEndpoint, bytesConstRef(m_recvData.data(), _len));
210 doRead();
211 }));
212}

214template <typename Handler, unsigned MaxDatagramSize>
215void UDPSocket<Handler, MaxDatagramSize>::doWrite() {
216 if (m_closed) return;
217
218 const UDPDatagram& datagram = m_sendQ[0];
219 auto self(UDPSocket<Handler, MaxDatagramSize>::shared_from_this());
220 bi::udp::endpoint endpoint(datagram.endpoint());
221 m_socket.async_send_to(boost::asio::buffer(datagram.data), endpoint,
222 [this, self, endpoint](boost::system::error_code _ec, std::size_t) {
223 if (m_closed) return disconnectWithError(_ec);
224
225 if (_ec != boost::system::errc::success)
226 net_logger_->debug("Failed delivering UDP message. {} : {}", _ec.value(), _ec.message());
227
228 Guard l(x_sendQ);
229 m_sendQ.pop_front();
230 if (m_sendQ.empty()) return;
231 doWrite();
232 });
233}

235template <typename Handler, unsigned MaxDatagramSize>
236void UDPSocket<Handler, MaxDatagramSize>::disconnectWithError(boost::system::error_code _ec) {
237 // If !started and already stopped, shutdown has already occurred. (EOF or
238 // Operation canceled)
239 if (!m_started && m_closed && !m_socket.is_open() /* todo: veirfy this logic*/) return;
240
241 assert(_ec);
242 {
243 // disconnect-operation following prior non-zero errors are ignored
244 Guard l(x_socketError);
245 if (m_socketError != boost::system::error_code()) return;
246 m_socketError = _ec;
247 }
248 // TODO: (if non-zero error) schedule high-priority writes
249
250 // prevent concurrent disconnect
251 bool expected = true;
252 if (!m_started.compare_exchange_strong(expected, false)) return;
253
254 // set m_closed to true to prevent undeliverable egress messages
255 bool wasClosed = m_closed;
256 m_closed = true;
257
258 // close sockets
259 boost::system::error_code ec;
260 m_socket.shutdown(bi::udp::socket::shutdown_both, ec);
261 m_socket.close();
262
263 // socket never started if it never left stopped-state (pre-handshake)
264 if (wasClosed) return;
265
266 m_host.onSocketDisconnected(this);
267}
