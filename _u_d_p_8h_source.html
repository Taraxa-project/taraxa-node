157 template <typename Handler, unsigned MaxDatagramSize>
158 void UDPSocket<Handler, MaxDatagramSize>::connect() {
159  bool expect = false;
160  if (!m_started.compare_exchange_strong(expect, true)) return;
161 
162  m_socket.open(bi::udp::v4());
163  try {
164  m_socket.bind(m_endpoint);
165  } catch (...) {
166  m_socket.bind(bi::udp::endpoint(bi::udp::v4(), m_endpoint.port()));
167  }
168 
169  // clear write queue so reconnect doesn't send stale messages
170  Guard l(x_sendQ);
171  m_sendQ.clear();
172 
173  m_closed = false;
174  doRead();
175 }

177 template <typename Handler, unsigned MaxDatagramSize>
178 bool UDPSocket<Handler, MaxDatagramSize>::send(UDPDatagram const& _datagram) {
179  if (m_closed) return false;
180 
181  Guard l(x_sendQ);
182  m_sendQ.push_back(_datagram);
183  if (m_sendQ.size() == 1) doWrite();
184 
185  return true;
186 }

188 template <typename Handler, unsigned MaxDatagramSize>
189 void UDPSocket<Handler, MaxDatagramSize>::doRead() {
190  if (m_closed) return;
191 
192  auto self(UDPSocket<Handler, MaxDatagramSize>::shared_from_this());
193  m_socket.async_receive_from(
194  boost::asio::buffer(m_recvData), m_recvEndpoint,
195  boost::asio::bind_executor(strand_, [this, self](boost::system::error_code _ec, size_t _len) {
196  if (m_closed) return disconnectWithError(_ec);
197 
198  if (_ec != boost::system::errc::success)
199  cnetlog << "Receiving UDP message failed. " << _ec.value() << " : " << _ec.message();
200 
201  if (_len) m_host.onPacketReceived(this, m_recvEndpoint, bytesConstRef(m_recvData.data(), _len));
202  doRead();
203  }));
204 }

206 template <typename Handler, unsigned MaxDatagramSize>
207 void UDPSocket<Handler, MaxDatagramSize>::doWrite() {
208  if (m_closed) return;
209 
210  const UDPDatagram& datagram = m_sendQ[0];
211  auto self(UDPSocket<Handler, MaxDatagramSize>::shared_from_this());
212  bi::udp::endpoint endpoint(datagram.endpoint());
213  m_socket.async_send_to(boost::asio::buffer(datagram.data), endpoint,
214  [this, self, endpoint](boost::system::error_code _ec, std::size_t) {
215  if (m_closed) return disconnectWithError(_ec);
216 
217  if (_ec != boost::system::errc::success)
218  cnetlog << "Failed delivering UDP message. " << _ec.value() << " : " << _ec.message();
219 
220  Guard l(x_sendQ);
221  m_sendQ.pop_front();
222  if (m_sendQ.empty()) return;
223  doWrite();
224  });
225 }

227 template <typename Handler, unsigned MaxDatagramSize>
228 void UDPSocket<Handler, MaxDatagramSize>::disconnectWithError(boost::system::error_code _ec) {
229  // If !started and already stopped, shutdown has already occured. (EOF or
230  // Operation canceled)
231  if (!m_started && m_closed && !m_socket.is_open() /* todo: veirfy this logic*/) return;
232 
233  assert(_ec);
234  {
235  // disconnect-operation following prior non-zero errors are ignored
236  Guard l(x_socketError);
237  if (m_socketError != boost::system::error_code()) return;
238  m_socketError = _ec;
239  }
240  // TODO: (if non-zero error) schedule high-priority writes
241 
242  // prevent concurrent disconnect
243  bool expected = true;
244  if (!m_started.compare_exchange_strong(expected, false)) return;
245 
246  // set m_closed to true to prevent undeliverable egress messages
247  bool wasClosed = m_closed;
248  m_closed = true;
249 
250  // close sockets
251  boost::system::error_code ec;
252  m_socket.shutdown(bi::udp::socket::shutdown_both, ec);
253  m_socket.close();
254 
255  // socket never started if it never left stopped-state (pre-handshake)
256  if (wasClosed) return;
257 
258  m_host.onSocketDisconnected(this);
259 }
