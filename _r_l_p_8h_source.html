308 template <class _T = unsigned>
309 std::enable_if_t<is_integer<_T>(Signedness::any), _T> toInt(int _flags = Strict) const {
310 requireGood();
311 if constexpr (std::numeric_limits<std::decay_t<_T>>::is_signed) {
312 if (isList()) {
313 if ((_flags & Strict) && itemCountStrict() != 1) {
314 BOOST_THROW_EXCEPTION(BadCast());
315 }
316 return -operator[](0).template toInt<_T>(_flags);
317 }
318 }
319 if ((!isInt() && !(_flags & AllowNonCanon)) || isList() || isNull()) {
320 if (_flags & ThrowOnFail)
321 BOOST_THROW_EXCEPTION(BadCast());
322 else
323 return 0;
324 }
325
326 auto p = payload();
327 if (p.size() > intTraits<_T>::maxSize && (_flags & FailIfTooBig)) {
328 if (_flags & ThrowOnFail)
329 BOOST_THROW_EXCEPTION(BadCast());
330 else
331 return 0;
332 }
333
334 return fromBigEndian<_T>(p);
335 }

337 int64_t toPositiveInt64(int _flags = Strict) const {
338 int64_t i = toInt<int64_t>(_flags);
339 if ((_flags & ThrowOnFail) && i < 0) BOOST_THROW_EXCEPTION(BadCast());
340 return i;

343 template <class _N>
344 _N toHash(int _flags = Strict) const {
345 requireGood();
346 auto p = payload();
347 auto l = p.size();
348 if (!isData() || (l > _N::size && (_flags & FailIfTooBig)) || (l < _N::size && (_flags & FailIfTooSmall))) {
349 if (_flags & ThrowOnFail)
350 BOOST_THROW_EXCEPTION(BadCast());
351 else
352 return _N();
353 }
354
355 _N ret;
356 size_t s = std::min<size_t>(_N::size, l);
357 memcpy(ret.data() + _N::size - s, p.data(), s);
358 return ret;
359 }

517 if (!_i)
518 m_out.push_back(c_rlpDataImmLenStart);
519 else if (_i < c_rlpDataImmLenStart)
520 m_out.push_back((::byte)_i);
521 else {
522 unsigned br = bytesRequired(_i);

526 auto brbr = bytesRequired(br);

528 BOOST_THROW_EXCEPTION(RLPException() << errinfo_comment("Number too large for RLP"));
529 m_out.push_back((::byte)(c_rlpDataIndLenZero + brbr));
530 pushInt(br, brbr);
531 }
532 pushInt(_i, br);
533 }
534 noteAppended();
535 return *this;
536 }
