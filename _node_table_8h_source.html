33 void processEvents() {
34 std::list<std::pair<NodeID, NodeTableEventType>> events;
35 {
36 Guard l(x_events);
37 if (!m_nodeEventHandler.size()) return;
38 m_nodeEventHandler.unique();
39 for (auto const& n : m_nodeEventHandler) events.push_back(std::make_pair(n, m_events[n]));
40 m_nodeEventHandler.clear();
41 m_events.clear();
42 }
43 for (auto const& e : events) callback_(e.first, e.second);
44 }

48 Guard l(x_events);
49 m_nodeEventHandler.push_back(_n);
50 m_events[_n] = _e;
51 }

425struct DiscoveryDatagram : public RLPXDatagramFace {
426 static constexpr std::chrono::seconds c_timeToLiveS{60};
427
429 DiscoveryDatagram(bi::udp::endpoint const& _to) : RLPXDatagramFace(_to), expiration(futureFromEpoch(c_timeToLiveS)) {}
430
432 DiscoveryDatagram(bi::udp::endpoint const& _from, NodeID const& _fromid, h256 const& _echo)
433 : RLPXDatagramFace(_from), sourceid(_fromid), echo(_echo) {}
434
435 // These two are set for inbound packets only.
436 NodeID sourceid; // sender public key (from signature)
437 h256 echo; // hash of encoded packet, for reply tracking

458struct PingNode : DiscoveryDatagram {
459 using DiscoveryDatagram::DiscoveryDatagram;
460 PingNode(NodeIPEndpoint const& _src, NodeIPEndpoint const& _dest, unsigned _chain_id)
461 : DiscoveryDatagram(_dest), chain_id(_chain_id), source(_src), destination(_dest) {}
462 PingNode(bi::udp::endpoint const& _from, NodeID const& _fromid, h256 const& _echo)
463 : DiscoveryDatagram(_from, _fromid, _echo) {}

465 static constexpr uint8_t type = 1;
466 uint8_t packetType() const override { return type; }
467
468 unsigned version = 0;
469 unsigned chain_id = 0;
470 NodeIPEndpoint source;
471 NodeIPEndpoint destination;
472 std::optional<uint64_t> seq;
473
474 void streamRLP(RLPStream& _s) const override {
475 _s.appendList(seq.has_value() ? 6 : 5);
476 _s << dev::p2p::c_protocolVersion;
477 _s << chain_id;
478 source.streamRLP(_s);
479 destination.streamRLP(_s);
480 _s << *expiration;
481 if (seq.has_value()) _s << *seq;
482 }
483
484 void interpretRLP(bytesConstRef _bytes) override {
485 RLP r(_bytes, RLP::AllowNonCanon | RLP::ThrowOnFail);
486 version = r[0].toInt<unsigned>();
487 chain_id = r[1].toInt<unsigned>();
488 source.interpretRLP(r[2]);
489 destination.interpretRLP(r[3]);
490 expiration = r[4].toInt<uint32_t>();
491 if (r.itemCount() > 5 && r[5].isInt()) seq = r[5].toInt<uint64_t>();
492 }
493
494 std::string typeName() const override { return "Ping"; }
495};

580 struct Neighbour {
581 Neighbour(Node const& _node) : endpoint(_node.get_endpoint()), node(_node.id) {
582 // For external node we need to replace udp to reported one as we can communicate on upd port that's not available
583 // to everyone

585 endpoint.setUdpPort(_node.external_udp_port);

592 _s.appendList(4);
593 endpoint.streamRLP(_s, NodeIPEndpoint::StreamInline);

595 }

601 std::vector<Neighbour> neighbours;
602
603 void streamRLP(RLPStream& _s) const override {
604 _s.appendList(2);
605 _s.appendList(neighbours.size());
606 for (auto const& n : neighbours) n.streamRLP(_s);
607 _s << *expiration;
608 }
