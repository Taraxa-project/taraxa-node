49 bool processVote(const std::shared_ptr<PbftVote>& vote, const std::shared_ptr<PbftBlock>& pbft_block,
50 const std::shared_ptr<TaraxaPeer>& peer, bool validate_max_round_step) {
51 if (pbft_block && !validateVoteAndBlock(vote, pbft_block)) {
52 throw MaliciousPeerException("Received vote's voted value != received pbft block");
53 }
54
55 if (vote_mgr_->voteInVerifiedMap(vote)) {
56 LOG(this->log_dg_) << "Vote " << vote->getHash() << " already inserted in verified queue";
57 return false;
58 }
59
60 // Validate vote's period, round and step min/max values
61 if (const auto vote_valid = validateVotePeriodRoundStep(vote, peer, validate_max_round_step); !vote_valid.first) {
62 LOG(this->log_wr_) << "Vote period/round/step " << vote->getHash()
63 << " validation failed. Err: " << vote_valid.second;
64 return false;
65 }
66
67 // Check is vote is unique per period, round & step & voter -> each address can generate just 1 vote
68 // (for a value that isn't NBH) per period, round & step
69 if (auto vote_valid = vote_mgr_->isUniqueVote(vote); !vote_valid.first) {
70 // Create double voting proof
71 slashing_manager_->submitDoubleVotingProof(vote, vote_valid.second);
72 throw MaliciousPeerException("Received double vote", vote->getVoter());
73 }
74
75 // Validate vote's signature, vrf, etc...
76 if (const auto vote_valid = vote_mgr_->validateVote(vote); !vote_valid.first) {
77 LOG(this->log_wr_) << "Vote " << vote->getHash() << " validation failed. Err: " << vote_valid.second;
78 return false;
79 }
80
81 if (!vote_mgr_->addVerifiedVote(vote)) {
82 LOG(this->log_dg_) << "Vote " << vote->getHash() << " already inserted in verified queue(race condition)";
83 return false;
84 }
85
86 if (pbft_block) {
87 pbft_mgr_->processProposedBlock(pbft_block, vote);
88 }
89
90 return true;
91 }

98 bool isPbftRelevantVote(const std::shared_ptr<PbftVote>& vote) const {
99 const auto [current_pbft_round, current_pbft_period] = pbft_mgr_->getPbftRoundAndPeriod();
100
101 if (vote->getPeriod() >= current_pbft_period && vote->getRound() >= current_pbft_round) {
102 // Standard current or future vote
103 return true;
104 } else if (vote->getPeriod() == current_pbft_period && vote->getRound() == (current_pbft_round - 1) &&
105 vote->getType() == PbftVoteTypes::next_vote) {
106 // Previous round next vote
107 return true;
108 } else if (vote->getPeriod() == current_pbft_period - 1 && vote->getType() == PbftVoteTypes::cert_vote) {
109 // Previous period cert vote - potential reward vote
110 return true;
111 }
112
113 return false;
114 }

116 virtual void sendPbftVotesBundle(const std::shared_ptr<TaraxaPeer>& peer,
117 std::vector<std::shared_ptr<PbftVote>>&& votes) {
118 if (votes.empty()) {
119 return;
120 }
121
122 auto sendVotes = [this, &peer](std::vector<std::shared_ptr<PbftVote>>&& votes) {
123 auto packet = VotesBundlePacket{OptimizedPbftVotesBundle{.votes = std::move(votes)}};
124 if (this->sealAndSend(peer->getId(), SubprotocolPacketType::kVotesBundlePacket, encodePacketRlp(packet))) {
125 LOG(this->log_dg_) << " Votes bundle with " << packet.votes_bundle.votes.size() << " votes sent to "
126 << peer->getId();
127 for (const auto& vote : packet.votes_bundle.votes) {
128 peer->markPbftVoteAsKnown(vote->getHash());
129 }
130 }
131 };
132
133 if (votes.size() <= kMaxVotesInBundleRlp) {
134 sendVotes(std::move(votes));
135 return;
136 } else {
137 // Need to split votes into multiple packets
138 size_t index = 0;
139 while (index < votes.size()) {
140 const size_t votes_count = std::min(kMaxVotesInBundleRlp, votes.size() - index);
141
142 const auto begin_it = std::next(votes.begin(), index);
143 const auto end_it = std::next(begin_it, votes_count);
144
145 std::vector<std::shared_ptr<PbftVote>> votes_sub_vector;
146 std::move(begin_it, end_it, std::back_inserter(votes_sub_vector));
147
148 sendVotes(std::move(votes_sub_vector));
149
150 index += votes_count;
151 }
152 }
153 }

164 std::pair<bool, std::string> validateVotePeriodRoundStep(const std::shared_ptr<PbftVote>& vote,
165 const std::shared_ptr<TaraxaPeer>& peer,
166 bool validate_max_round_step) {
167 const auto [current_pbft_round, current_pbft_period] = pbft_mgr_->getPbftRoundAndPeriod();
168
169 auto genErrMsg = [period = current_pbft_period, round = current_pbft_round,
170 step = pbft_mgr_->getPbftStep()](const std::shared_ptr<PbftVote>& vote) -> std::string {
171 std::stringstream err;
172 err << "Vote " << vote->getHash() << " (period, round, step) = (" << vote->getPeriod() << ", " << vote->getRound()
173 << ", " << vote->getStep() << "). Current PBFT (period, round, step) = (" << period << ", " << round << ", "
174 << step << ")";
175 return err.str();
176 };
177
178 // Period validation
179 // vote->getPeriod() == current_pbft_period - 1 && cert_vote -> potential reward vote
180 if (vote->getPeriod() < current_pbft_period - 1 ||
181 (vote->getPeriod() == current_pbft_period - 1 && vote->getType() != PbftVoteTypes::cert_vote)) {
182 return {false, "Invalid period(too small): " + genErrMsg(vote)};
183 } else if (this->kConf.network.ddos_protection.vote_accepting_periods &&
184 vote->getPeriod() - 1 >
185 current_pbft_period + this->kConf.network.ddos_protection.vote_accepting_periods) {
186 // skip this check if kConf.network.ddos_protection.vote_accepting_periods == 0
187 // vote->getPeriod() - 1 is here because votes are validated against vote_period - 1 in dpos contract
188 // Do not request round sync too often here
189 if (vote->getVoter() == peer->getId() &&
190 std::chrono::system_clock::now() - last_pbft_block_sync_request_time_ > kSyncRequestInterval) {
191 // request PBFT chain sync from this node
192 this->sealAndSend(
193 peer->getId(), SubprotocolPacketType::kGetPbftSyncPacket,
194 encodePacketRlp(GetPbftSyncPacket{std::max(vote->getPeriod() - 1, peer->pbft_chain_size_.load())}));
195 last_pbft_block_sync_request_time_ = std::chrono::system_clock::now();
196 }
197
198 return {false, "Invalid period(too big): " + genErrMsg(vote)};
199 }
200
201 // Round validation
202 auto checking_round = current_pbft_round;
203 // If period is not the same we assume current round is equal to 1
204 // So we won't accept votes for future period with round bigger than kConf.network.vote_accepting_steps
205 if (current_pbft_period != vote->getPeriod()) {
206 checking_round = 1;
207 }
208
209 // vote->getRound() == checking_round - 1 && next_vote -> previous round next vote
210 if (vote->getRound() < checking_round - 1 ||
211 (vote->getRound() == checking_round - 1 && vote->getType() != PbftVoteTypes::next_vote)) {
212 return {false, "Invalid round(too small): " + genErrMsg(vote)};
213 } else if (validate_max_round_step && this->kConf.network.ddos_protection.vote_accepting_rounds &&
214 vote->getRound() >= checking_round + this->kConf.network.ddos_protection.vote_accepting_rounds) {
215 // skip this check if kConf.network.vote_accepting_rounds == 0
216 // Trigger votes(round) syncing only if we are in sync in terms of period
217 if (current_pbft_period == vote->getPeriod()) {
218 // Do not request round sync too often here
219 if (vote->getVoter() == peer->getId() &&
220 std::chrono::system_clock::now() - last_votes_sync_request_time_ > kSyncRequestInterval) {
221 // request round votes sync from this node
222 this->requestPbftNextVotesAtPeriodRound(peer->getId(), current_pbft_period, current_pbft_round);
223 last_votes_sync_request_time_ = std::chrono::system_clock::now();
224 }
225 }
226
227 return {false, "Invalid round(too big): " + genErrMsg(vote)};
228 }
229
230 // Step validation
231 auto checking_step = pbft_mgr_->getPbftStep();
232 // If period or round is not the same we assume current step is equal to 1
233 // So we won't accept votes for future rounds with step bigger than kConf.network.vote_accepting_steps
234 if (current_pbft_period != vote->getPeriod() || current_pbft_round != vote->getRound()) {
235 checking_step = 1;
236 }
237
238 // skip check if kConf.network.vote_accepting_steps == 0
239 if (validate_max_round_step && this->kConf.network.ddos_protection.vote_accepting_steps &&
240 vote->getStep() >= checking_step + this->kConf.network.ddos_protection.vote_accepting_steps) {
241 return {false, "Invalid step(too big): " + genErrMsg(vote)};
242 }
243
244 return {true, ""};
245 }
