40 void startSyncingPbft() {
41 if (pbft_syncing_state_->isPbftSyncing()) {
42 LOG(this->log_dg_) << "startSyncingPbft called but syncing_ already true";
43 return;
44 }
45
46 std::shared_ptr<TaraxaPeer> peer = getMaxChainPeer();
47 if (!peer) {
48 LOG(this->log_nf_) << "Restarting syncing PBFT not possible since no connected peers";
49 return;
50 }
51
52 auto pbft_sync_period = pbft_mgr_->pbftSyncingPeriod();
53 if (peer->pbft_chain_size_ > pbft_sync_period) {
54 auto peer_id = peer->getId().abridged();
55 auto peer_pbft_chain_size = peer->pbft_chain_size_.load();
56 if (!pbft_syncing_state_->setPbftSyncing(true, pbft_sync_period, std::move(peer))) {
57 LOG(this->log_dg_) << "startSyncingPbft called but syncing_ already true";
58 return;
59 }
60 LOG(this->log_si_) << "Restarting syncing PBFT from peer " << peer_id << ", peer PBFT chain size "
61 << peer_pbft_chain_size << ", own PBFT chain synced at period " << pbft_sync_period;
62
63 if (syncPeerPbft(pbft_sync_period + 1)) {
64 // Disable snapshots only if are syncing from scratch
65 if (pbft_syncing_state_->isDeepPbftSyncing()) {
66 db_->disableSnapshots();
67 }
68 } else {
69 pbft_syncing_state_->setPbftSyncing(false);
70 }
71 } else {
72 LOG(this->log_nf_) << "Restarting syncing PBFT not needed since our pbft chain size: " << pbft_sync_period << "("
73 << pbft_chain_->getPbftChainSize() << ")"
74 << " is greater or equal than max node pbft chain size:" << peer->pbft_chain_size_;
75 db_->enableSnapshots();
76 }

86 bool syncPeerPbft(PbftPeriod request_period) {
87 const auto syncing_peer = pbft_syncing_state_->syncingPeer();
88 if (!syncing_peer) {
89 LOG(this->log_er_) << "Unable to send GetPbftSyncPacket. No syncing peer set.";
90 return false;
91 }
92
93 if (request_period > syncing_peer->pbft_chain_size_) {
94 LOG(this->log_wr_) << "Invalid syncPeerPbft argument. Node " << syncing_peer->getId() << " chain size "
95 << syncing_peer->pbft_chain_size_ << ", requested period " << request_period;
96 return false;
97 }
98
99 LOG(this->log_nf_) << "Send GetPbftSyncPacket with period " << request_period << " to node "
100 << syncing_peer->getId();

102 encodePacketRlp(GetPbftSyncPacket{request_period}));

110 void requestPendingDagBlocks(std::shared_ptr<TaraxaPeer> peer = nullptr) {
111 if (!peer) {
112 peer = getMaxChainPeer([](const std::shared_ptr<TaraxaPeer> &peer) {
113 if (peer->peer_dag_synced_ || !peer->dagSyncingAllowed()) {
114 return false;
115 }
116 return true;
117 });
118 if (!peer) {
119 LOG(this->log_nf_) << "requestPendingDagBlocks not possible since no peers are matching conditions";
120 return;
121 }
122 }
123
124 if (!peer) {
125 LOG(this->log_nf_) << "requestPendingDagBlocks not possible since no connected peers";
126 return;
127 }
128
129 // This prevents ddos requesting dag blocks. We can only request this one time from one peer.
130 if (peer->peer_dag_synced_) {
131 LOG(this->log_nf_) << "requestPendingDagBlocks not possible since already requested for peer";
132 return;
133 }
134
135 // Only request dag blocks if periods are matching
136 auto pbft_sync_period = pbft_mgr_->pbftSyncingPeriod();
137 if (pbft_sync_period == peer->pbft_chain_size_) {
138 // This prevents parallel requests
139 if (bool b = false; !peer->peer_dag_syncing_.compare_exchange_strong(b, !b)) {
140 LOG(this->log_nf_) << "requestPendingDagBlocks not possible since already requesting for peer";
141 return;
142 }
143 LOG(this->log_nf_) << "Request pending blocks from peer " << peer->getId();
144 std::vector<blk_hash_t> known_non_finalized_blocks;
145 auto [period, blocks] = dag_mgr_->getNonFinalizedBlocks();
146 for (auto &level_blocks : blocks) {
147 for (auto &block : level_blocks.second) {
148 known_non_finalized_blocks.emplace_back(block);
149 }
150 }
151
152 requestDagBlocks(peer->getId(), std::move(known_non_finalized_blocks), period);
153 }

156 std::shared_ptr<TaraxaPeer> getMaxChainPeer(std::function<bool(const std::shared_ptr<TaraxaPeer> &)> filter_func =
157 [](const std::shared_ptr<TaraxaPeer> &) { return true; }) {
158 std::shared_ptr<TaraxaPeer> max_pbft_chain_peer;
159 PbftPeriod max_pbft_chain_size = 0;
160 uint64_t max_node_dag_level = 0;
161
162 // Find peer with max pbft chain and dag level
163 for (auto const &peer : this->peers_state_->getAllPeers()) {
164 // Apply the filter function
165 if (!filter_func(peer.second)) {
166 continue;
167 }
168
169 if (peer.second->pbft_chain_size_ > max_pbft_chain_size) {
170 if (peer.second->peer_light_node &&
171 pbft_mgr_->pbftSyncingPeriod() + peer.second->peer_light_node_history < peer.second->pbft_chain_size_) {
172 LOG(this->log_er_) << "Disconnecting from light node peer " << peer.first
173 << " History: " << peer.second->peer_light_node_history
174 << " chain size: " << peer.second->pbft_chain_size_;
175 this->disconnect(peer.first, dev::p2p::UserReason);
176 continue;
177 }
178 max_pbft_chain_size = peer.second->pbft_chain_size_;
179 max_node_dag_level = peer.second->dag_level_;
180 max_pbft_chain_peer = peer.second;
181 } else if (peer.second->pbft_chain_size_ == max_pbft_chain_size && peer.second->dag_level_ > max_node_dag_level) {
182 max_node_dag_level = peer.second->dag_level_;
183 max_pbft_chain_peer = peer.second;
184 }
185 }
186 return max_pbft_chain_peer;
187 }
