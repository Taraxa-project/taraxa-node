129  // This prevents ddos requesting dag blocks. We can only request this one time from one peer.

140  LOG(this->log_nf_) << "requestPendingDagBlocks not possible since already requesting for peer";
141  return;
142  }
143  LOG(this->log_nf_) << "Request pending blocks from peer " << peer->getId();
144  std::vector<blk_hash_t> known_non_finalized_blocks;
145  auto [period, blocks] = dag_mgr_->getNonFinalizedBlocks();
146  for (auto &level_blocks : blocks) {
