240  template <unsigned P, unsigned M>
241  inline FixedHash<M> bloomPart() const {
242  unsigned const c_bloomBits = M * 8;
243  unsigned const c_mask = c_bloomBits - 1;
244  unsigned const c_bloomBytes = (StaticLog2<c_bloomBits>::result + 7) / 8;
245 
246  static_assert((M & (M - 1)) == 0, "M must be power-of-two");
247  static_assert(P * c_bloomBytes <= N, "out of range");
248 
249  FixedHash<M> ret;
250  ::byte const* p = data();
251  for (unsigned i = 0; i < P; ++i) {
252  unsigned index = 0;
253  for (unsigned j = 0; j < c_bloomBytes; ++j, ++p) index = (index << 8) | *p;
254  index &= c_mask;
255  ret[M - 1 - index / 8] |= (1 << (index % 8));
256  }
257  return ret;
258  }

263  unsigned ret = 0;

266  for (;; ++ret, d <<= 1) {

268  }
269  } else
270  ret += 8;
271  return ret;
272  }
