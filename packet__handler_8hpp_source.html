56 void processPacket(const threadpool::PacketData& packet_data) override {
57 try {
58 const auto begin = std::chrono::steady_clock::now();
59
60 // It can rarely happen that packet was received and pushed into the queue when peer was still in peers map,
61 // in the meantime the connection was lost and we started to process packet from such peer
62 const auto peer = peers_state_->getPacketSenderPeer(packet_data.from_node_id_, packet_data.type_);
63 if (!peer.first) [[unlikely]] {
64 LOG(log_wr_) << "Unable to process packet. Reason: " << peer.second;
65 disconnect(packet_data.from_node_id_, dev::p2p::UserReason);
66 return;
67 }
68
69 // Decode packet rlp into packet object
70 auto packet = decodePacketRlp<PacketType>(packet_data.rlp_);
71
72 // Main processing function
73 process(std::move(packet), peer.first);
74
75 auto processing_duration =
76 std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - begin);
77 auto tp_wait_duration = std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() -
78 packet_data.receive_time_);
79
80 PacketStats packet_stats{1 /* count */, packet_data.rlp_.data().size(), processing_duration, tp_wait_duration};
81 peer.first->addSentPacket(packet_data.type_str_, packet_stats);
82
83 if (kConf.network.ddos_protection.log_packets_stats) {
84 packets_stats_->addReceivedPacket(packet_data.type_str_, packet_data.from_node_id_, packet_stats);
85 }
86
87 } catch (const MaliciousPeerException& e) {
88 // thrown during packets processing -> malicious peer, invalid rlp items count, ...
89 // If there is custom peer set in exception, disconnect him, not packet sender
90 if (const auto custom_peer = e.getPeer(); custom_peer.has_value()) {
91 handle_caught_exception(e.what(), packet_data, *custom_peer, e.getDisconnectReason(),
92 true /* set peer as malicious */);
93 } else {
94 handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_, e.getDisconnectReason(),
95 true /* set peer as malicious */);
96 }
97 } catch (const PacketProcessingException& e) {
98 // thrown during packets processing...
99 handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_, e.getDisconnectReason(),
100 true /* set peer as malicious */);
101 } catch (const dev::RLPException& e) {
102 // thrown during parsing inside aleth/libdevcore -> type mismatch
103 handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_, dev::p2p::DisconnectReason::BadProtocol,
104 true /* set peer as malicious */);
105 } catch (const std::exception& e) {
106 handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_);
107 } catch (...) {
108 handle_caught_exception("Unknown exception", packet_data, packet_data.from_node_id_);
109 }
110 }

142 auto host = peers_state_->host_.lock();
143 if (!host) {
144 LOG(log_er_) << "sealAndSend failed to obtain host";
145 return false;
146 }
147
148 if (const auto peer = peers_state_->getPacketSenderPeer(node_id, packet_type); !peer.first) [[unlikely]] {
149 LOG(log_wr_) << "Unable to send packet. Reason: " << peer.second;
150 host->disconnect(node_id, dev::p2p::UserReason);
151 return false;
152 }
153
154 const auto begin = std::chrono::steady_clock::now();
155 const size_t packet_size = rlp_bytes.size();
156
157 host->send(node_id, TARAXA_CAPABILITY_NAME, packet_type, std::move(rlp_bytes),
158 [begin, node_id, packet_size, packet_type, this]() {
159 if (!kConf.network.ddos_protection.log_packets_stats) {
160 return;
161 }
162
163 PacketStats packet_stats{
164 1 /* count */, packet_size,
165 std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - begin),
166 std::chrono::microseconds{0}};
167
168 packets_stats_->addSentPacket(convertPacketTypeToString(packet_type), node_id, packet_stats);
169 });
