60inline std::string toString(std::chrono::time_point<T> const& _e, std::string const& _format = "%F %T")
61#endif
62{
63 unsigned long milliSecondsSinceEpoch =
64 std::chrono::duration_cast<std::chrono::milliseconds>(_e.time_since_epoch()).count();
65 auto const durationSinceEpoch = std::chrono::milliseconds(milliSecondsSinceEpoch);
66 std::chrono::time_point<std::chrono::system_clock> const tpAfterDuration(durationSinceEpoch);
67
68 tm timeValue;
69 auto time = std::chrono::system_clock::to_time_t(tpAfterDuration);
70#if defined(_WIN32)
71 gmtime_s(&timeValue, &time);
72#else
73 gmtime_r(&time, &timeValue);
74#endif
75
76 unsigned const millisRemainder = milliSecondsSinceEpoch % 1000;
77 char buffer[1024];
78 if (strftime(buffer, sizeof(buffer), _format.c_str(), &timeValue))
79 return std::string(buffer) + "." +
80 (millisRemainder < 1 ? "000"
81 : millisRemainder < 10 ? "00"
82 : millisRemainder < 100 ? "0"
83 : "") +
84 std::to_string(millisRemainder) + "Z";
85 return std::string();
86}
