27template <class Iterator>
28std::string toHex(Iterator _it, Iterator _end, std::string const& _prefix) {
29 typedef std::iterator_traits<Iterator> traits;
30 static_assert(sizeof(typename traits::value_type) == 1, "toHex needs byte-sized element type");
31
32 static char const* hexdigits = "0123456789abcdef";
33 size_t off = _prefix.size();
34 std::string hex(std::distance(_it, _end) * 2 + off, '0');
35 hex.replace(0, off, _prefix);
36 for (; _it != _end; _it++) {
37 hex[off++] = hexdigits[(*_it >> 4) & 0x0f];
38 hex[off++] = hexdigits[*_it & 0x0f];
39 }
40 return hex;
41}

104template <class T, class Out>
105inline void toBigEndian(T _val, Out& o_out) {
106 static_assert(std::is_same<bigint, T>::value || !std::numeric_limits<T>::is_signed,
107 "only unsigned types or bigint supported"); // bigint does not carry sign
108 // bit on shift
109 for (auto i = o_out.size(); i != 0; _val >>= 8, i--) {
110 T v = _val & (T)0xff;
111 o_out[i - 1] = (typename Out::value_type)(uint8_t)v;
112 }
113}

119template <class T, class _In>
120inline T fromBigEndian(_In const& _bytes) {
121 T ret = (T)0;
122 for (auto i : _bytes) ret = (T)((ret << 8) | (::byte)(typename std::make_unsigned<decltype(i)>::type)i);
123 return ret;
124}

152 static_assert(std::is_same<bigint, T>::value || !std::numeric_limits<T>::is_signed,
153 "only unsigned types or bigint supported"); // bigint does not carry sign

218template <class T>
219void trimFront(T& _t, unsigned _elements) {
220 static_assert(std::is_standard_layout<typename T::value_type>::value, "");
221 memmove(_t.data(), _t.data() + _elements, (_t.size() - _elements) * sizeof(_t[0]));
222 _t.resize(_t.size() - _elements);
223}

227template <class T, class _U>
228void pushFront(T& _t, _U _e) {
229 static_assert(std::is_standard_layout<typename T::value_type>::value, "");
230 _t.push_back(_e);
231 memmove(_t.data() + 1, _t.data(), (_t.size() - 1) * sizeof(_e));
232 _t[0] = _e;
233}

333static constexpr std::enable_if_t<!std::is_enum_v<N>, bool> is_integer(Signedness sig) {
334 using limits_t = std::numeric_limits<std::decay_t<N>>;
335 return limits_t::is_integer &&
