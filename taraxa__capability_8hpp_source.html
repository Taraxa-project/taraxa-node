143template <typename PacketHandlerType>
144std::shared_ptr<PacketHandlerType> TaraxaCapability::getSpecificHandler(SubprotocolPacketType packet_type) const {
145 // Note: Allow to manually cast only to known base classes types.
146 // We support multiple taraxa capabilities, which can contain different versions of packet handlers and casting
147 // directly to final classes types breaks the functionality...
148 switch (packet_type) {
149 case SubprotocolPacketType::kPbftSyncPacket:
150 case SubprotocolPacketType::kStatusPacket:
151 if (!std::is_same<ISyncPacketHandler, PacketHandlerType>::value) {
152 assert(false);
153 }
154 break;
155
156 case SubprotocolPacketType::kTransactionPacket:
157 if (!std::is_same<ITransactionPacketHandler, PacketHandlerType>::value) {
158 assert(false);
159 }
160 break;
161
162 case SubprotocolPacketType::kVotePacket:
163 case SubprotocolPacketType::kVotesBundlePacket:
164 if (!std::is_same<IVotePacketHandler, PacketHandlerType>::value) {
165 assert(false);
166 }
167 break;
168
169 case SubprotocolPacketType::kPillarVotePacket:
170 if (!std::is_same<IPillarVotePacketHandler, PacketHandlerType>::value) {
171 assert(false);
172 }
173 break;
174
175 case SubprotocolPacketType::kGetPillarVotesBundlePacket:
176 if (!std::is_same<IGetPillarVotesBundlePacketHandler, PacketHandlerType>::value) {
177 assert(false);
178 }
179 break;
180
181 case SubprotocolPacketType::kDagBlockPacket:
182 if (!std::is_same<IDagBlockPacketHandler, PacketHandlerType>::value) {
183 assert(false);
184 }
185 break;
186
187 default:
188 assert(false);
189 return nullptr;
190 }
191
192 auto handler = packets_handlers_->getSpecificHandler(packet_type);
193 return std::dynamic_pointer_cast<PacketHandlerType>(handler);
194}
