24template <class I>
25class ServerInterface {
26 public:
27 using MethodPointer = AbstractMethodPointer<I>;
28 using NotificationPointer = AbstractNotificationPointer<I>;
29
30 using MethodBinding = std::tuple<jsonrpc::Procedure, AbstractMethodPointer<I>>;
31 using NotificationBinding = std::tuple<jsonrpc::Procedure, AbstractNotificationPointer<I>>;
32 using Methods = std::vector<MethodBinding>;
33 using Notifications = std::vector<NotificationBinding>;
34 struct RPCModule {
35 std::string name;

40 virtual ~ServerInterface() {}

42 Notifications const& notifications() const { return m_notifications; }
45 virtual RPCModules implementedModules() const = 0;
46
47 protected:
48 void bindAndAddMethod(jsonrpc::Procedure const& _proc, MethodPointer _pointer) {
49 m_methods.emplace_back(_proc, _pointer);

52 m_notifications.emplace_back(_proc, _pointer);

111template <class I, class... Is>
112class ModularServer<I, Is...> : public ModularServer<Is...> {
113 public:
114 using MethodPointer = AbstractMethodPointer<I>;
115 using NotificationPointer = AbstractNotificationPointer<I>;
116
117 ModularServer(std::shared_ptr<I> _i, std::shared_ptr<Is>... _is) : ModularServer<Is...>(_is...), m_interface(_i) {
118 if (!m_interface) return;
119 for (auto const& method : m_interface->methods()) {
120 m_methods[std::get<0>(method).GetProcedureName()] = std::get<1>(method);
121 this->m_handler->AddProcedure(std::get<0>(method));
122 }
123
124 for (auto const& notification : m_interface->notifications()) {
125 m_notifications[std::get<0>(notification).GetProcedureName()] = std::get<1>(notification);
126 this->m_handler->AddProcedure(std::get<0>(notification));
127 }
128 // Store module with version.
129 for (auto const& module : m_interface->implementedModules())
130 this->m_implementedModules[module.name] = module.version;
131 }
132
133 virtual void HandleMethodCall(jsonrpc::Procedure& _proc, Json::Value const& _input, Json::Value& _output) override {
134 auto pointer = m_methods.find(_proc.GetProcedureName());
135 if (pointer != m_methods.end()) {
136 try {
137 (m_interface.get()->*(pointer->second))(_input, _output);
138 } catch (Json::Exception const& ex) {
139 throw jsonrpc::JsonRpcException(jsonrpc::Errors::ERROR_RPC_INVALID_PARAMS, ex.what());
140 }
141 } else
142 ModularServer<Is...>::HandleMethodCall(_proc, _input, _output);
143 }
144
145 virtual void HandleNotificationCall(jsonrpc::Procedure& _proc, Json::Value const& _input) override {
146 auto pointer = m_notifications.find(_proc.GetProcedureName());
147 if (pointer != m_notifications.end()) {
148 try {
149 (m_interface.get()->*(pointer->second))(_input);
150 } catch (Json::Exception const& ex) {
151 throw jsonrpc::JsonRpcException(jsonrpc::Errors::ERROR_RPC_INVALID_PARAMS, ex.what());
152 }
153 } else
154 ModularServer<Is...>::HandleNotificationCall(_proc, _input);
155 }
156
157 private:
158 std::shared_ptr<I> m_interface;
159 std::map<std::string, MethodPointer> m_methods;
160 std::map<std::string, NotificationPointer> m_notifications;
161};
