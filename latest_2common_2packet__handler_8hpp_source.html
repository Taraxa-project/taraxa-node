64  LOG(log_wr_) << "Unable to process packet. Reason: " << peer.second;

95  true /* set peer as malicious */);

100  true /* set peer as malicious */);

103  handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_, dev::p2p::DisconnectReason::BadProtocol,
104  true /* set peer as malicious */);
105  } catch (const std::exception& e) {
106  handle_caught_exception(e.what(), packet_data, packet_data.from_node_id_);
107  } catch (...) {
